const int buttonPin=2;const int green=8;const int yellow=9;const int red=10;int time=3000;int yellowtime=1000;void setup(){Serial.begin(9600);pinMode(green,OUTPUT);pinMode(yellow,OUTPUT);pinMode(red,OUTPUT);pinMode(buttonPin,INPUT);attachInterrupt(digitalPinToInterrupt(buttonPin),Button_pressed,RISING);digitalWrite(red,HIGH);}void loop(){if(digitalRead(red)==HIGH){digitalWrite(red,LOW);digitalWrite(green,HIGH);Serial.println("Green");delay(time);}else if(digitalRead(green)==HIGH){digitalWrite(green,LOW);digitalWrite(yellow,HIGH);Serial.println("Yellow");delay(yellowtime);}else if(digitalRead(yellow)==HIGH){digitalWrite(yellow,LOW);digitalWrite(red,HIGH);Serial.println("Red");delay(time);}}void Button_pressed(){if(digitalRead(green)==HIGH){digitalWrite(green,LOW);digitalWrite(yellow,HIGH);Serial.println("Yellow");delay(yellowtime);}}
